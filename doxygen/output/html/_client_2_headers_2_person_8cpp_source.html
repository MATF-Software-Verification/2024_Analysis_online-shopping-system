168bool Person::consistency(const Date& DOB, const int& myAge)
169{
170 // Determining current date/time based on current system
171 // 'now' now stores the number of seconds elapsed since Jan 1, 1970
172 time_t now = time(0);
173
174 // Returns a pointer to the tm structure (defined in ctime)
175 // representing local time and date
176
177 tm* t = localtime(&now);
178 // t->tm_mday stores day, t->tm_mon stores month (from 0 to 11) and t->tm_year stores years since 1900
179
180 // Calculating age
181 int age;
182 if (DOB.month < (t->tm_mon + 1))
183 age = (t->tm_year + 1900) - DOB.year;
184 else if (DOB.month > (t->tm_mon + 1))
185 age = (t->tm_year + 1900) - DOB.year - 1; // Person's birthday this year is yet to arrive
186 else
187 {
188 if (DOB.day > t->tm_mday)
189 age = (t->tm_year + 1900) - DOB.year - 1; // Person's birthday this year is yet to arrive
190 else
191 age = (t->tm_year + 1900) - DOB.year;
192 }
193
194 // Comparing age
195 if (age == myAge)
196 return true;
197
198 cout << "Date of birth is not consistent with entered age.\n";
199 return false;
